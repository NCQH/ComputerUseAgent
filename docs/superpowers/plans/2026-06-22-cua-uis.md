# CUA UIs Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Wire the Plan 1–3 pieces into a runnable app with two frontends — a CLI (prompt_toolkit) and a desktop GUI (PySide6) — that share one `AgentSession`, stream events, accept real-time steering, and surface confirmation prompts.

**Architecture:** Keep the integration logic in small, fully-tested, UI-agnostic helpers: a pure `format_event` renderer, a `build_session` assembler (provider + executor + gate + bus + queue → `AgentSession`, with `display_size` validation), a confirm-handler factory, and a `SessionRunner` that owns the asyncio run-loop lifecycle (including restarting `run()` when the user submits after the agent goes idle). The two UI shells (CLI, GUI) are thin I/O layers that lazy-import their toolkit, subscribe to the bus via `format_event`, and drive a `SessionRunner` — they are reviewed by inspection and excluded from the offline suite (like the Dockerfile and real-SDK factory branches in earlier plans).

**Tech Stack:** Python 3.11+, asyncio, pytest, pytest-asyncio. CLI uses prompt_toolkit; GUI uses PySide6 + qasync — both imported lazily.

## Global Constraints

- Python 3.11+; asyncio.
- Reuse existing pieces unchanged: `build_provider`/`build_executor` (`cua/config.py`), `IrreversibilityGate` (`cua/core/safety.py`), `EventBus` + event types (`cua/core/events.py`), `InputQueue` (`cua/core/queue.py`), `AgentSession` (`cua/core/session.py`), `ConfirmRequest` (`cua/models.py`).

- UI-agnostic helpers (`format_event`, `build_session`, confirm factory, `SessionRunner`) are fully unit-tested offline with fakes — no prompt_toolkit / PySide6 / qasync / real SDK / Docker needed to run the suite.

- `do()`/provider/executor never-raise contracts are already guaranteed by Plans 1–3; UIs must not add escape paths.
- `display_size` must be validated at the assembly boundary: a 2-tuple of positive ints, else `ValueError`.
- Files focused, <400 lines.
- pytest + pytest-asyncio, `asyncio_mode="auto"`.
- Package import root is `cua`.

**Mapping (exact):**
- `StateChanged(state)` → `f"[state] {state}"`
- `StepCompleted(action, result)` → `f"[step] {type(action).__name__} -> ok"` if `result.success` else `f"[step] {type(action).__name__} -> FAIL: {result.error}"`
- `ScreenshotTaken(_)` → `None`
- `LogMessage(text)` → `text`
- `ConfirmRequested(request)` → `f"[confirm] {request.reason} :: {request.action}"`
- `ErrorOccurred(message)` → `f"[error] {message}"`
- anything else → `None`

# cua/ui/format.py
"""Pure rendering of bus events into one-line log strings (UI-agnostic)."""
from __future__ import annotations

from cua.core.events import (
    StateChanged, StepCompleted, ScreenshotTaken, LogMessage, ConfirmRequested, ErrorOccurred,
)


def format_event(event) -> str | None:
    if isinstance(event, StateChanged):
        return f"[state] {event.state}"
    if isinstance(event, StepCompleted):
        name = type(event.action).__name__
        if event.result.success:
            return f"[step] {name} -> ok"
        return f"[step] {name} -> FAIL: {event.result.error}"
    if isinstance(event, ScreenshotTaken):
        return None
    if isinstance(event, LogMessage):
        return event.text
    if isinstance(event, ConfirmRequested):
        return f"[confirm] {event.request.reason} :: {event.request.action}"
    if isinstance(event, ErrorOccurred):
        return f"[error] {event.message}"
    return None
```

# cua/app.py
"""Assemble a wired AgentSession from provider/executor names + injected clients."""
from __future__ import annotations

from cua.config import build_provider, build_executor
from cua.core.events import EventBus
from cua.core.queue import InputQueue
from cua.core.safety import IrreversibilityGate
from cua.core.session import AgentSession


def validate_display_size(display_size) -> None:
    if not isinstance(display_size, (tuple, list)) or len(display_size) != 2:
        raise ValueError(f"display_size must be a 2-tuple, got {display_size!r}")
    w, h = display_size
    if not (isinstance(w, int) and isinstance(h, int)) or w <= 0 or h <= 0:
        raise ValueError(f"display_size must be positive ints, got {display_size!r}")


def build_session(
    provider_name,
    executor_name,
    *,
    confirm_handler,
    provider_client=None,
    page=None,
    http_client=None,
    display_size=(1280, 800),
    denylist=None,
    max_steps=50,
) -> AgentSession:
    validate_display_size(display_size)
    provider = build_provider(provider_name, client=provider_client, display_size=display_size)
    if executor_name.strip().lower() == "web":
        executor = build_executor("web", page=page, display_size=display_size)
    else:
        executor = build_executor(executor_name, client=http_client, display_size=display_size)
    gate = IrreversibilityGate(denylist=denylist)
    bus = EventBus()
    queue = InputQueue()
    return AgentSession(
        provider=provider,
        executor=executor,
        gate=gate,
        bus=bus,
        queue=queue,
        confirm_handler=confirm_handler,
        max_steps=max_steps,
    )
```

import asyncio
from cua.ui.runner import SessionRunner


class OneShotSession:
    """run() returns immediately; each call counts."""
    def __init__(self):
        self.submitted = []
        self.runs = 0
    async def submit(self, text):
        self.submitted.append(text)
    async def run(self):
        self.runs += 1


class BlockingSession:
    """run() blocks on an event so we can observe a long-running agent."""
    def __init__(self, release):
        self.release = release
        self.submitted = []
        self.runs = 0
    async def submit(self, text):
        self.submitted.append(text)
    async def run(self):
        self.runs += 1
        await self.release.wait()


async def test_first_submit_starts_a_run():
    s = OneShotSession()
    r = SessionRunner(s)
    await r.submit("hi")
    await r.aclose()

    assert s.runs == 1

    s = OneShotSession()
    r = SessionRunner(s)
    await r.submit("a")
    await r.aclose()
    await r.submit("b")
    await r.aclose()
    assert s.runs == 2
    assert s.submitted == ["a", "b"]

    r = SessionRunner(s)

# cua/ui/runner.py
"""Owns the agent run-loop lifecycle so UIs can submit and resume cleanly."""
from __future__ import annotations

import asyncio


class SessionRunner:
    def __init__(self, session) -> None:
        self.session = session
        self._task: asyncio.Task | None = None

    @property
    def is_running(self) -> bool:
        return self._task is not None and not self._task.done()

    async def submit(self, text: str) -> None:
        await self.session.submit(text)
        if not self.is_running:
            self._task = asyncio.create_task(self.session.run())

    async def aclose(self) -> None:
        if self._task is not None:
            await self._task
            self._task = None
```

# cua/ui/cli.py
"""CLI frontend: prompt_toolkit input loop + event log. I/O shell (lazy import)."""
from __future__ import annotations

from cua.ui.format import format_event
from cua.ui.runner import SessionRunner


async def run_cli(session, runner=None) -> None:
    # Lazy import so the package + test suite do not require prompt_toolkit.
    from prompt_toolkit import PromptSession
    from prompt_toolkit.patch_stdout import patch_stdout

    runner = runner or SessionRunner(session)

    def on_event(event) -> None:
        line = format_event(event)
        if line is not None:
            print(line)

    session.bus.subscribe(on_event)

    pt = PromptSession()
    print("CUA CLI — gõ yêu cầu rồi Enter. Ctrl-D để thoát.")
    with patch_stdout():
        while True:
            try:
                text = await pt.prompt_async("> ")
            except (EOFError, KeyboardInterrupt):
                break
            text = text.strip()
            if not text:
                continue
            await runner.submit(text)
    await runner.aclose()
```

# cua/ui/gui.py
"""GUI frontend: PySide6 window + qasync event loop. I/O shell (lazy import)."""
from __future__ import annotations

from cua.ui.format import format_event
from cua.ui.runner import SessionRunner


def run_gui(session, build_confirm_handler=None) -> None:
    # Lazy imports so the package + test suite do not require PySide6/qasync.
    import asyncio
    import sys
    import qasync
    from PySide6.QtWidgets import (
        QApplication, QWidget, QVBoxLayout, QHBoxLayout, QTextEdit, QLineEdit,
        QPushButton, QMessageBox,
    )

    app = QApplication.instance() or QApplication(sys.argv)
    loop = qasync.QEventLoop(app)
    asyncio.set_event_loop(loop)

    window = QWidget()
    window.setWindowTitle("CUA")
    layout = QVBoxLayout(window)
    log = QTextEdit()
    log.setReadOnly(True)
    layout.addWidget(log)
    row = QHBoxLayout()
    box = QLineEdit()
    send = QPushButton("Gửi")
    row.addWidget(box)
    row.addWidget(send)
    layout.addLayout(row)

    def on_event(event) -> None:
        line = format_event(event)
        if line is not None:
            log.append(line)

    session.bus.subscribe(on_event)

    # Confirm handler bound to a modal dialog (used if the entrypoint did not
    # supply one). build_confirm_handler(window) lets the entrypoint customize it.
    if build_confirm_handler is not None:
        session.confirm_handler = build_confirm_handler(window)
    else:
        async def _confirm(request) -> bool:
            answer = QMessageBox.question(
                window, "Xác nhận", f"{request.reason}\n\n{request.action}\n\nCho phép?",
            )
            return answer == QMessageBox.StandardButton.Yes
        session.confirm_handler = _confirm

    runner = SessionRunner(session)

    def submit_text() -> None:
        text = box.text().strip()
        if text:
            box.clear()
            asyncio.ensure_future(runner.submit(text))

    send.clicked.connect(submit_text)
    box.returnPressed.connect(submit_text)

    window.resize(900, 600)
    window.show()
    with loop:
        loop.run_forever()
```

# cua/__main__.py
"""CLI/GUI entrypoint: select UI, provider, executor and launch."""
from __future__ import annotations

import argparse

from cua.app import build_session
from cua.ui.confirm import auto_approve


def parse_args(argv=None) -> argparse.Namespace:
    parser = argparse.ArgumentParser(prog="cua", description="Computer-Use Agent")
    parser.add_argument("--ui", choices=["cli", "gui"], default="cli")
    parser.add_argument("--provider", choices=["claude", "openai"], default="claude")
    parser.add_argument("--executor", choices=["web", "desktop"], default="desktop")
    parser.add_argument("--width", type=int, default=1280)
    parser.add_argument("--height", type=int, default=800)
    return parser.parse_args(argv)


def main(argv=None) -> None:
    args = parse_args(argv)
    display_size = (args.width, args.height)

    # Real clients/page/container are constructed lazily by the factories when
    # no injected client is passed. The web executor needs a launched Playwright
    # page; for the desktop executor the factory builds an httpx client.
    session = build_session(
        args.provider,
        args.executor,
        confirm_handler=auto_approve,  # replaced by the UI's interactive handler below
        display_size=display_size,
    )

    if args.ui == "gui":
        from cua.ui.gui import run_gui
        run_gui(session)
    else:
        import asyncio
        from cua.ui.cli import run_cli
        from cua.ui.confirm import make_cli_confirm_handler

        # Bind the confirm handler to stdin so prompts share the terminal.
        async def ask() -> str:
            return await asyncio.get_event_loop().run_in_executor(None, input)

        session.confirm_handler = make_cli_confirm_handler(print, ask)
        asyncio.run(run_cli(session))


if __name__ == "__main__":
    main()
```

Note: the `--executor web` path requires a launched Playwright page, which `build_executor("web", page=None)` raises on by design (Plan 3) — wiring a real browser/context is left to the operator/runtime and is out of this plan's offline scope. `--executor desktop` builds a real httpx client via the factory and talks to a running sandbox container.
